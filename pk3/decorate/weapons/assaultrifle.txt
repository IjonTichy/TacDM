actor "Assault Rifle": TacDMWeapon
{
    Weapon.SelectionOrder 800
    Weapon.AmmoUse 1
    Weapon.AmmoUse2 1
    Weapon.AmmoGive 30
    Weapon.AmmoType1 "AssaultBullet"
    Weapon.AmmoType2 "AssaultBullet"

    Inventory.PickupMessage "Standard-issue assault rifle"
    Obituary "%k gunned down %o."

    Weapon.SlotNumber 4

    +NOAUTOFIRE

    States
    {
      Ready:
        ASLT A 1 A_WeaponReady
        Loop
      Deselect:
        ASLT A 1 A_Lower
        Loop

      Select:
        ASLT A 0 A_JumpIfInventory("IsBot", 1, 1)
        goto TrueSelect
        ASLT A 0 A_GiveInventory("Assault~Rifle", 1)
        ASLT A 0 A_TakeInventory("Assault Rifle", 1)
        ASLT A 0 A_SelectWeapon ("Assault~Rifle")
        stop

      TrueSelect:
        ASLT A 1 A_Raise
        Loop

      Fire:
        ASLT B 0 bright A_FireBullets(1, 1, -1, 15, "BulletPuff")
        ASLT B 1 bright A_PlaySound("tacticaldm/assaultriflefire", CHAN_WEAPON)
        ASLT C 2 bright
        ASLT D 2
        Goto Ready
      AltFire:
        ASLT B 0 bright A_FireBullets(3, 3, -1, 15, "BulletPuff")
        ASLT B 1 bright A_PlaySound("tacticaldm/assaultriflefire", CHAN_WEAPON)
        ASLT C 1 bright
        ASLT B 0 bright A_FireBullets(3, 3, -1, 15, "BulletPuff")
        ASLT B 1 bright A_PlaySound("tacticaldm/assaultriflefire", CHAN_WEAPON)
        ASLT C 1 bright
        ASLT B 0 bright A_FireBullets(3, 3, -1, 15, "BulletPuff")
        ASLT B 1 bright A_PlaySound("tacticaldm/assaultriflefire", CHAN_WEAPON)
        ASLT C 2 bright
        ASLT D 2
        ASLT A 6
        Goto Ready
      Spawn:
        ASLT Z -1
        Stop
    }
}
