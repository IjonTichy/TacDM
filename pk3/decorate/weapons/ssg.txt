actor "Double-Barrel Shotgun": TacDMWeapon
{
    Weapon.SelectionOrder 100
    Weapon.AmmoUse1 2
    Weapon.AmmoUse2 2
    Weapon.AmmoGive1 2
    Weapon.AmmoGive2 0
    Weapon.AmmoType1 "Buckshot"
    Weapon.AmmoType2 "Slug"

    Weapon.SlotNumber 4

    DamageType TacDMPhysical
    Weapon.PreferredSkin "T-SSG"

    Inventory.PickupMessage "Double-barrel shotgun"
    Obituary "%k splattered %o with a super shotgun."
    States
    {
      Spawn:
        SGN2 A -1
        Stop
      Ready:
        PKS2 A 1 A_WeaponReady
        Loop
      Deselect:
        PKS2 A 1 A_Lower
        Loop
      Select:
        PKS2 A 1 A_Raise
        Loop
      Fire:
        PKS2 A 3
        PKS2 A 0 A_FireBullets(10, 8, 30, 5, "TacDMBulletPuff")
        PKS2 A 0 A_PlaySound ("weapons/sshotf", CHAN_WEAPON)
        PKS2 A 1 bright A_GunFlash
        PKS2 A 0 bright A_Checkreload
        PKS2 C 1
        PKS2 D 2
        PKS2 E 12
        PKS2 F 5 A_OpenShotgun2
        PKS2 G 3
        PKS2 H 4
        PKS2 IJ 2
        PKS2 K 3
        PKS2 L 2 A_LoadShotgun2
        PKS2 M 3
        PKS2 NOPQ 2
        PKS2 R 3 A_CloseShotgun2
        PKS2 S 3
        PKS2 A 4
        PKS2 A 0 A_ReFire
        goto Ready

      AltFire:
        PKS2 A 3
        PKS2 A 0 A_FireBullets (4, 1, 2, 40, "TacDMBulletPuff")
        PKS2 A 0 A_PlaySound ("weapons/sshotf", CHAN_WEAPON)
        PKS2 A 1 bright A_GunFlash
        PKS2 A 0 A_CheckReload
        PKS2 C 1
        PKS2 D 2
        PKS2 E 16
        PKS2 F 5 A_OpenShotgun2
        PKS2 G 3
        PKS2 H 4
        PKS2 IJ 2
        PKS2 K 3
        PKS2 L 2 A_LoadShotgun2
        PKS2 M 3
        PKS2 NOPQ 2
        PKS2 R 3 A_CloseShotgun2
        PKS2 S 3
        PKS2 A 4
        PKS2 A 0 A_ReFire
        goto Ready
      Flash:
        SHT2 IJ 1 Bright A_Light1
        goto LightDone
    }
}
