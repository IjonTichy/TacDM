actor "Double-Barrel Shotgun": TacDMWeapon replaces SuperShotgun
{
    Weapon.SelectionOrder 100
    Weapon.AmmoUse1 2
    Weapon.AmmoUse2 2
    Weapon.AmmoGive1 2
    Weapon.AmmoGive2 0
    Weapon.AmmoType1 "Buckshot"
    Weapon.AmmoType2 "Slug"

    Weapon.SlotNumber 3

    Inventory.PickupMessage "Double-barrel shotgun"
    Obituary "%k splattered %o with a super shotgun."
    States
    {
      Spawn:
        SGN2 A -1
        Stop
      Ready:
        PKS2 A 1 A_WeaponReady
        Loop
      Deselect:
        PKS2 A 1 A_Lower
        Loop
      Select:
        PKS2 A 1 A_Raise
        Loop
      Fire:
        PKS2 A 3
        PKS2 A 0 A_FireBullets (11.2, 7.1, 20, 7, "BulletPuff")
        PKS2 A 0 A_PlaySound ("weapons/sshotf", CHAN_WEAPON)
        PKS2 A 1 bright A_GunFlash
        PKS2 A 3 bright A_Checkreload
        goto GenericReload

      AltFire:
        PKS2 A 3
        PKS2 A 0 A_FireBullets (5, 3, 2, 35, "BulletPuff")
        PKS2 A 0 A_PlaySound ("weapons/sshotf", CHAN_WEAPON)
        PKS2 A 1 bright A_GunFlash
        PKS2 A 7 bright A_Checkreload
        goto GenericReload

      GenericReload:
        PKS2 B 2
        PKS2 C 3
        PKS2 D 4
        PKS2 E 4
        PKS2 F 2 A_OpenShotgun2
        PKS2 GHIJ 2
        PKS2 K 3
        PKS2 L 2 A_LoadShotgun2
        PKS2 M 3
        PKS2 NOPQ 2
        PKS2 R 3 A_PlaySound("weapons/sshotc", CHAN_WEAPON)
        PKS2 S 2
        PKS2 T 1 A_ReFire
        Goto Ready
      Flash:
        SHT2 J 3 Bright A_Light2
        SHT2 I 2 Bright A_Light1
        Goto LightDone
    }
}
